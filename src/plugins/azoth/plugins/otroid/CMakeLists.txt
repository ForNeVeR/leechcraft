CMAKE_MINIMUM_REQUIRED (VERSION 2.8)
PROJECT (leechcraft_azoth_otroid)
INCLUDE (InitLCPlugin OPTIONAL)

INCLUDE (${QT_USE_FILE})

SET (CMAKE_MODULE_PATH "${CMAKE_CURRENT_SOURCE_DIR}/cmake")

FIND_PACKAGE (LibOTR REQUIRED)
MESSAGE (STATUS "Found OTR: ${LIBOTR_LIBRARY}")

INCLUDE_DIRECTORIES (${AZOTH_INCLUDE_DIR}
	${CMAKE_CURRENT_BINARY_DIR}
	${LIBOTR_INCLUDE_DIR}
	${LEECHCRAFT_INCLUDE_DIR}
	${LEECHCRAFT_INCLUDE_DIR}/azoth
	)
SET (OTROID_SRCS
	otroid.cpp
	)
SET (OTROID_HEADERS
	otroid.h
	)
SET (OTROID_RESOURCES azothotroidresources.qrc)
CreateTrVars ("azoth_otroid" "en;ru_RU;uk_UA" OTROID_TRANSLATIONS OTROID_COMPILED_TRANSLATIONS)
IF (NOT LC_NO_MOC)
	QT4_WRAP_CPP (OTROID_MOC_SRCS ${OTROID_HEADERS})
ENDIF (NOT LC_NO_MOC)
QT4_ADD_RESOURCES (OTROID_RCCS ${OTROID_RESOURCES})

ADD_CUSTOM_COMMAND (OUTPUT
	${OTROID_COMPILED_TRANSLATIONS}
	COMMAND "${QT_LRELEASE_EXECUTABLE}" ${OTROID_TRANSLATIONS}
	DEPENDS ${OTROID_TRANSLATIONS}
	WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
	)
ADD_LIBRARY (leechcraft_azoth_otroid SHARED
	${OTROID_COMPILED_TRANSLATIONS}
	${OTROID_SRCS}
	${OTROID_MOC_SRCS}
	${OTROID_RCCS}
	)
TARGET_LINK_LIBRARIES (leechcraft_azoth_otroid
	${QT_LIBRARIES}
	${LEECHCRAFT_LIBRARIES}
	${LIBOTR_LIBRARY}
	)
INSTALL (TARGETS leechcraft_azoth_otroid DESTINATION ${LC_PLUGINS_DEST})
#INSTALL (FILES azothotroidsettings.xml DESTINATION ${LC_SETTINGS_DEST})
INSTALL (FILES ${OTROID_COMPILED_TRANSLATIONS} DESTINATION ${LC_TRANSLATIONS_DEST})
