IF(NOT QT_USE_FILE)
	CMAKE_MINIMUM_REQUIRED(VERSION 2.6)
	IF(COMMAND cmake_policy)
		cmake_policy(SET CMP0003 NEW)
	ENDIF(COMMAND cmake_policy)

	PROJECT(leechcraft_gmailnotifier)

	IF(NOT CMAKE_MODULE_PATH)
		SET(CMAKE_MODULE_PATH "/usr/local/share/leechcraft/cmake;/usr/share/leechcraft/cmake")
	ENDIF(NOT CMAKE_MODULE_PATH)

	FIND_PACKAGE(Boost REQUIRED)
	FIND_PACKAGE(Qt4 REQUIRED)
	FIND_PACKAGE(LeechCraft REQUIRED)
ENDIF(NOT QT_USE_FILE)

SET(QT_USE_QTNETWORK TRUE)
SET(QT_USE_QTXML TRUE)

INCLUDE(${QT_USE_FILE})
INCLUDE_DIRECTORIES(
	${CMAKE_CURRENT_BINARY_DIR}
	${CMAKE_CURRENT_SOURCE_DIR}
	${Boost_INCLUDE_DIR}
	${LEECHCRAFT_INCLUDE_DIR}
)
SET(HEADERS
	gmailnotifier.h
	xmlsettingsmanager.h
	gmailchecker.h
)
SET(SRCS
	gmailnotifier.cpp
	xmlsettingsmanager.cpp
	gmailchecker.cpp
)
SET(RESOURCES
	res/resources.qrc
)
CreateTrVars("gmailnotifier" "de;en;fr;it;ru_RU;uk_UA" TRANSLATIONS COMPILED_TRANSLATIONS)

IF (NOT LC_NO_MOC)
	QT4_WRAP_CPP(MOC_SRCS ${HEADERS})
ENDIF (NOT LC_NO_MOC)
QT4_ADD_RESOURCES(RCCS ${RESOURCES})

ADD_CUSTOM_COMMAND(OUTPUT
	${COMPILED_TRANSLATIONS}
	COMMAND "${QT_LRELEASE_EXECUTABLE}" ${TRANSLATIONS}
	DEPENDS ${TRANSLATIONS}
	WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
)

ADD_LIBRARY(leechcraft_gmailnotifier SHARED
	${COMPILED_TRANSLATIONS}
	${SRCS}
	${RCCS}
	${MOC_SRCS}
)
TARGET_LINK_LIBRARIES(leechcraft_gmailnotifier
	${QT_LIBRARIES}
	${LEECHCRAFT_LIBRARIES}
)
INSTALL(TARGETS leechcraft_gmailnotifier DESTINATION ${LC_PLUGINS_DEST})
INSTALL(FILES ${COMPILED_TRANSLATIONS} DESTINATION ${LC_TRANSLATIONS_DEST})
INSTALL(FILES gmailnotifiersettings.xml DESTINATION ${LC_SETTINGS_DEST})
